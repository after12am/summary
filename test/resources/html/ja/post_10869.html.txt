function appendRecentEntries(){
    $.ajax({
        type: "GET",
        dataType: "json",
        url: "/common/json/pr_article.json"+"?"+(new Date().getTime()),
        success: function(data){
            var $cxRec = $("#cxRecentries");
            var $cxRecList = $cxRec.find("li");
            var cxRecLen = $cxRec.find("li").length;
            var prLen = data.article.length;
            var prAr = [];
            var articleNum = prLen < 3 ? prLen : 3;
            for(var i=0; i<prLen; i++){
                prAr[i] = i;
            }
            prAr = prAr.sort(function() {
                return Math.random() - Math.random();
            });
            for(var i=0; i<articleNum; i++){
                var r = Math.floor( Math.random() * cxRecLen );
                $cxRecList.eq(r).before(data.article[prAr[i]]);
            }
        },
        error : function(data) {}
    });
}
